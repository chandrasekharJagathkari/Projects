# Function to handle repeated characters
def handle_repeated_chars(text):
    cleaned_text = ""
    prev_char = ""
    
    for char in text:
        if char != prev_char:
            cleaned_text += char
        prev_char = char
    
    return cleaned_text

# Perform sentiment analysis with TextBlob for each sentiment column

        df[f'{col} Sentiment (TextBlob)'] = df[col].apply(lambda x: TextBlob(x).sentiment.polarity)
